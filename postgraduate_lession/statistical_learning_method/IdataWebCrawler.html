import re
from bs4 import BeautifulSoup
from urllib.request import urlopen
from urllib import parse  # solve the Chinese word
import pandas as pd


def get_article_url(key_word_1, num_page_1):
url = 'https://search.ehn3.com/search?keyword={}&p={}'
for i in range(1, num_page_1):
yield url.format(key_word_1, i)


def get_information(paper_url):
html_paper = urlopen(paper_url)
soup_paper = BeautifulSoup(html_paper, features='lxml')

item_title = soup_paper.find('span', {"class": "headline"})
title_ = item_title.get_text().replace(' ', '').replace('\n', '')

item_abstract = soup_paper.find_all('v-card-text', {'style': "letter-spacing: 0.07em; line-height: 1.8;"})
abstract_ = item_abstract[0].get_text().replace(' ', '').replace('\n', '')

item = soup_paper.find_all('v-flex', {"lg1": "", "xs2": ""})
t = 1
if item[t].get_text().replace(' ', '').replace('\n', '') == '作者':
t = t + 1
author_ = item[t].get_text().replace(' ', '').replace('\n', '')
t = t + 1
else:
author_ = ''

if item[t].get_text().replace(' ', '').replace('\n', '') == '单位':
t = t + 1
institution_ = item[t].get_text().replace(' ', '').replace('\n', '')
t = t + 1
else:
institution_ = ''

if item[t].get_text().replace(' ', '').replace('\n', '') == '期刊':
t = t + 1
magazine_ = item[t].get_text().replace(' ', '').replace('\n', '')
t = t + 1
else:
magazine_ = ''

if item[t].get_text().replace(' ', '').replace('\n', '') == '时间':
t = t + 1
time_ = item[t].get_text().replace(' ', '').replace('\n', '')
t = t + 1
else:
time_ = ''

if item[t].get_text().replace(' ', '').replace('\n', '') == '关键词':
t = t + 1
key_words_ = item[t].get_text().replace(' ', '').replace('\n', '')
t = t + 1
else:
key_words_ = ''

if item[t].get_text().replace(' ', '').replace('\n', '') == '基金':
t = t + 1
founds_ = item[t].get_text().replace(' ', '').replace('\n', '')
t = t + 1
else:
founds_ = ''

return title_, abstract_, author_, institution_, magazine_, time_, key_words_, founds_


papers_url = []
titles = []
abstracts = []
authors = []
institutions = []
magazines = []
times = []
key_words = []
founds = []
key_word = '神经网络'
key_word = parse.quote(key_word)  #
num_page = 20

for i in get_article_url(key_word, num_page):
html = urlopen(i)
soup = BeautifulSoup(html, features='lxml')
sub_urls = soup.find_all("a", {"target": "_blank", "href": re.compile("/doc")})
for paper in sub_urls:
papers_url.append('https://search.ehn3.com' + paper['href'])
title, abstract, author, institution, magazine, time, key_word_, found_ = get_information(papers_url[-1])
print(title)
titles.append(title)
abstracts.append(abstract)
authors.append(author)
institutions.append(institution)
magazines.append(magazine)
times.append(time)
key_words.append(key_word_)
founds.append(found_)
a = pd.DataFrame(
columns=["title", "keyword", "abstract", "author", "institution", "time", "founds", "magazine", "link"])
a["link"] = papers_url
a["title"] = titles
a["keyword"] = key_words
a["abstract"] = abstracts
a["author"] = authors
a["institution"] = institutions
a["time"] = times
a["founds"] = founds
a["magazine"] = magazines
a.to_excel('jiafu_tang.xlsx', sheet_name='paper')
