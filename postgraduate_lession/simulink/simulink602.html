function [nP,dP] = paderm(tau,r,m)
%计算延迟项的pade有理近似
% tau=1;
% r=1;
% m=3;
c(1)=1;
for i=2:r+m+1
    c(i)=-c(i-1)*tau/(i-1);  %%%迭代
end
w=-c(r+2:m+r+1)';    % w 取r+2到m+r+1项的转置，取负,列向量
vv=[c(r+1:-1:1)';zeros(m-1-r,1)];  %vv由c的1到r+1项的转置，补0，列向量
W=rot90(hankel(c(m+r:-1:r+1),vv));  %rot90 逆时针旋转90度
V=rot90(hankel(c(r:-1:1)));  
%hankel(c)生成对称的Hankel矩阵。第一列元素为c，次对角线下方所有元素均为零。
%Hankel(c,r)矩阵的构成方式是：先指定第一列和最后一行，矩阵中其他所有元素都与其左下角相邻位置的元素相等
x=[1 (W\w)'];
y=[1 x(2:r+1)*V'+c(2:r+1)];
dP=x(m+1:-1:1)/x(m+1);
nP=y(r+1:-1:1)/x(m+1);
end
